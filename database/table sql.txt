CREATE TABLE `asset`.`asset_table` ( `id` INT NOT NULL AUTO_INCREMENT , `branch_name` VARCHAR(100) NOT NULL , `branch_code` INT NOT NULL , `date` DATE NOT NULL , `asset_type` VARCHAR(50) NOT NULL , `cat` VARCHAR(50) NOT NULL , `mf` VARCHAR(50) NOT NULL , `sno` VARCHAR(50) NOT NULL , `model` VARCHAR(25) NOT NULL , `cond` VARCHAR(20) NOT NULL , `remarks` TEXT NULL DEFAULT NULL , `image` BLOB NOT NULL , `hdd` VARCHAR(10) NULL DEFAULT NULL , `pro` VARCHAR(15) NULL DEFAULT NULL , `ram` VARCHAR(10) NULL DEFAULT NULL , `ip` VARCHAR(25) NULL DEFAULT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
